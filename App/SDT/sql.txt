CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    item_id INT,
    customer_id INT,
    comment TEXT,
    status VARCHAR(10),
    order_date DATE,
    FOREIGN KEY (item_id) REFERENCES merchandise(id),
    FOREIGN KEY (customer_id) REFERENCES clients(id)
);

CREATE TABLE clients (
    id INT PRIMARY KEY,
    name VARCHAR(255)
);

CREATE TABLE merchandise (
    id INT PRIMARY KEY,
    name VARCHAR(255),
    price DECIMAL(10, 2)
);
